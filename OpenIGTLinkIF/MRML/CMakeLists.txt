set(${KIT}_INCLUDE_DIRECTORIES
  ${Slicer_Base_INCLUDE_DIRS}
  ${CMAKE_CURRENT_SOURCE_DIR}/../Logic
  ${CMAKE_CURRENT_BINARY_DIR}/../Logic
  ${OpenIGTLinkIO_INCLUDE_DIRS}
  )

  vtkSlicerAnnotationsModuleMRML
